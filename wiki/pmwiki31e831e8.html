PHP: PregReplace

preg_replace()


<?PHP
$pattern[0] = '/\[\[tamad\]\]/';
$pattern[1] = '/\[\[sihirli_numara\]\]/';
$replacement[0] = "AdSoyad";
$replacement[1] = "123456789";
$subject = "Metin[[tamad]][[tamad]][[sihirli_numara]]";
$limit = "1"; //Sadece ilk buldu&#287;u de&#287;erleri de&#287;i&#351;tirir. Hepsi için "limit = -1"
$subject = preg_replace($pattern, $replacement, $subject, $limit);
echo $subject;
